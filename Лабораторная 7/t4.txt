/*4. Скорость первого автомобиля V1 км/ч, второго — V2 км/ч, расстояние между ними S км.
Определить расстояние между ними через T часов, если автомобили удаляются друг от друга. */
#include <iostream>
using namespace std;
int main()
{
    int v1=80,v2=90,s=50,t=3,p,dis;
    p=t*(v1+v2);
    dis=s+p;
    cout<<dis;
    return 0;
}
