/*4. Даны целые положительные числа A, B, C. На прямоугольнике размера A × B размещено
максимально возможное количество квадратов со стороной C (без наложений).
Найти количество квадратов, размещенных на прямоугольнике, а также площадь незанятой части прямоугольника.*/
#include <iostream>
#include <cmath>
using namespace std;
int main()
{
    int a,b,c,kol,s;
    cin >> a >> b >> c;
    kol = (a/c)*(b/c);
    s = a*b-kol*sqrt(c);
    cout << kol << endl;
    cout << s << endl;
    return 0;
}
