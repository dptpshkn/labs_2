/*4. Дано трехзначное число. 
Проверить истинность высказывания: 
«Цифры данного числа образуют возрастающую или убывающую последовательность».*/
#include <iostream>
using namespace std;

int main() {
 int A;
 cout << "A:";
 cin >> A;
 if ((A/100) > (A % 100 / 10) && (A % 100 / 10) > (A % 10) || (A / 100) < (A % 100 / 10) && (A % 100 / 10) < (A % 10)) {
 cout << "true";
 }
 else {
 cout << "false";
 }
}
