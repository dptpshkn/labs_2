/*1. Даны две переменные целого типа: A и B. Если их значения не равны,
то присвоить каждой переменной большее из этих значений,
а если равны, то присвоить переменным нулевые значения.
Вывести новые значения переменных A и B.*/

#include <iostream>
using namespace std;
 
int main() {

 int a, b;
 cout << "a:";
 cin >> a;
 cout << "b:";
 cin >> b;
 if (a==b) {
     a=0;
     b=0;
 } else {
     if (a > b) {
         b = a;
     } else {
         a = b;
     }
 }
 cout << a << endl;
 cout << b;
}
